import { getFirestore, collection, getDocs, doc, getDoc, updateDoc, increment } from "https://www.gstatic.com/firebasejs/10.12.2/firebase-firestore.js"; const db = getFirestore(); const shopItemsEl = document.getElementById('shop-items'); const starsCountEl = document.getElementById('stars-count'); const cashCountEl = document.getElementById('cash-count'); const usernameEl = document.getElementById('username'); let currentUser = null; // ---------- Load current user from localStorage ---------- async function loadCurrentUser() { const vipUser = JSON.parse(localStorage.getItem("vipUser")); if (!vipUser?.email) return; const uidKey = vipUser.email.replace(/[.#$[\]]/g, ','); // sanitize const userRef = doc(db, "users", uidKey); const snap = await getDoc(userRef); if (!snap.exists()) return; const data = snap.data(); currentUser = { uid: uidKey, chatId: data.chatId || "Guest", stars: data.stars || 0, cash: data.cash || 0 }; usernameEl.textContent = currentUser.chatId; starsCountEl.textContent = `${currentUser.stars} ⭐️`; cashCountEl.textContent = `₦${currentUser.cash}`; } // ---------- Render products ---------- function createProductCard(item) { const card = document.createElement('div'); card.classList.add('product-card'); card.innerHTML = `
${item.available} left
${item.name}
${item.cost} ⭐️
`; card.querySelector('.buy-btn').addEventListener('click', async () => { if (!currentUser) return alert("User not loaded"); if (currentUser.stars < item.cost) return alert("Not enough stars"); const userRef = doc(db, "users", currentUser.uid); await updateDoc(userRef, { stars: increment(-item.cost) }); currentUser.stars -= item.cost; starsCountEl.textContent = `${currentUser.stars} ⭐️`; // Record purchase await updateDoc(doc(collection(userRef, "purchases")), { itemId: item.id, name: item.name, cost: item.cost, timestamp: new Date() }); alert(`Purchased ${item.name}!`); }); return card; } // ---------- Load shop items ---------- async function loadShop() { const snap = await getDocs(collection(db, "ShopItems")); shopItemsEl.innerHTML = ''; snap.forEach(docSnap => { shopItemsEl.appendChild(createProductCard(docSnap.data())); }); } // ---------- Initialize ---------- window.addEventListener("DOMContentLoaded", async () => { await loadCurrentUser(); await loadShop(); document.getElementById('home-btn').addEventListener('click', () => { window.location.href = "index.html"; }); });
